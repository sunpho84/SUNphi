12  ///
13  /// Technically, the two IntSeq do not have to be ordered, the
14  /// insertion is done when HeadAft is larger than Ins. The elements
15  /// after are incremented by IncrAft. The insertion is ignored if
16  /// IgnoreIfPresent is true.

18  int IncrAft, // Increment after the found position
19  bool IgnoreIfPresent, // Ignore insertion if already present

23  DECLAUTO _insertInOrderedIntSeq(const IntSeq<Bef...>&,const IntSeq<HeadAft,TailAft...>&)
24  {
25  // Need to be inserted now
26  if constexpr(Ins<=HeadAft)
27  // Returns original IntSeq
28  if constexpr(Ins==HeadAft and IgnoreIfPresent)
29  return IntSeq<Bef...,HeadAft,TailAft...>{};
30  else
31  // Insert and shift
32  return IntSeq<Bef...,Ins,HeadAft+IncrAft,(TailAft+IncrAft)...>{};
33  else
34  // Move head to the left and nest
35  if constexpr(sizeof...(TailAft)>0)
36  return _insertInOrderedIntSeq<Ins,IncrAft,IgnoreIfPresent>(IntSeq<Bef...,HeadAft>{},IntSeq<TailAft...>{});
37  else
38  // Put at the end
39  return IntSeq<Bef...,HeadAft,Ins>{};
40  }
